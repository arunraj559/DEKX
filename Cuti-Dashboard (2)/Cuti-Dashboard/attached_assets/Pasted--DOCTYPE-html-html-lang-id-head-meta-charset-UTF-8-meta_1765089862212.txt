                <button class="btn-secondary" onclick="addApprover()" style="width: 100%; margin-bottom: 20px;">+ TAMBAH APPROVER</button>
                <div id="approversList" style="display: flex; flex-wrap: wrap; gap: 10px;">
                    <!-- Will be filled by JS -->
                </div>
            </div>
            <div class="modal-footer">
                <button class="btn-secondary" onclick="closeApproverModal()">BATAL</button>
                <button class="btn-primary" onclick="saveApprovers()">SIMPAN</button>

    <script>
        // ===== DATA MANAGEMENT =====
        const LEAVES_KEY = 'leave_data_v2';
        const APPROVERS_KEY = 'approvers_data';
        const USER_KEY = 'current_user';

        const DIVISION_LIMITS = {
            'KAPTEN': 3,
            'CS': 3,
            'KASIR': 3,
            'CS LINE': 1
        };

        let leaves = [];
        let approvers = [];
        let currentUser = null;

        // ===== INITIALIZATION =====
        function init() {
            loadData();
            currentUser = sessionStorage.getItem(USER_KEY);
            
            const today = new Date();
            document.getElementById('formMonth').valueAsDate = today;
            document.getElementById('viewMonth').valueAsDate = today;
            
            updateAllUI();
        }

        function loadData() {
            const stored = localStorage.getItem(LEAVES_KEY);
            leaves = stored ? JSON.parse(stored) : [];
            
            const approvStored = localStorage.getItem(APPROVERS_KEY);
            approvers = approvStored ? JSON.parse(approvStored) : [];
        }

        function saveData() {
            localStorage.setItem(LEAVES_KEY, JSON.stringify(leaves));
            localStorage.setItem(APPROVERS_KEY, JSON.stringify(approvers));
        }

        // ===== FORM SUBMISSION =====
        function submitLeave(e) {
            e.preventDefault();

            const month = document.getElementById('formMonth').value;
            const division = document.getElementById('division').value;
            const startDate = document.getElementById('startDate').value;
            const endDate = document.getElementById('endDate').value;

            // Check if slot is available
            const approvedInMonth = leaves.filter(l => l.month === month && l.division === division && l.approved);
            const limit = DIVISION_LIMITS[division];
            
            if (approvedInMonth.length >= limit) {
                showToast(`‚ùå SLOT PENUH - ${division} HANYA BOLEH ${limit} ORANG PER BULAN`);
                return;
            }

            const data = {
                id: Date.now().toString(),
                situs: document.getElementById('situs').value || '',
                perihal: document.getElementById('perihal').value,
                noPaspor: document.getElementById('noPaspor').value || '',
                namaStaff: document.getElementById('namaStaff').value,
                startDate: startDate,
                endDate: endDate,
                division: division,
                keterangan: document.getElementById('keterangan').value || '',
                kontakDarurat: document.getElementById('kontakDarurat').value || '',
                month: month,
                approved: false,
                approvedBy: null,
                approvedAt: null,
                createdAt: new Date().toISOString()
            };

            leaves.unshift(data);
            saveData();
            document.getElementById('leaveForm').reset();
            showToast('‚úÖ PENGAJUAN CUTI DISIMPAN');
            updateAllUI();
        }

        // ===== LEAVE ACTIONS =====
        function approveLea(id) {
            if (!currentUser || !approvers.includes(currentUser.toLowerCase())) {
                showToast('‚ùå ANDA TIDAK PUNYA HAK APPROVAL');
                openApproverModal();
                return;
            }

            const leave = leaves.find(l => l.id === id);
            if (!leave) return;

            const approvedInMonth = leaves.filter(l => 
                l.month === leave.month && 
                l.division === leave.division && 
                l.approved &&
                l.id !== id
            );
            const limit = DIVISION_LIMITS[leave.division];

            if (approvedInMonth.length >= limit) {
                showToast(`‚ùå SLOT PENUH - ${leave.division} SUDAH MENCAPAI LIMIT`);
                return;
            }

            leave.approved = true;
            leave.approvedBy = currentUser;
            leave.approvedAt = new Date().toISOString();
            saveData();
            showToast(`‚úÖ CUTI DISETUJUI - ${leave.namaStaff}`);
            updateAllUI();
        }

        function rejectLeave(id) {
            if (!currentUser || !approvers.includes(currentUser.toLowerCase())) {
                showToast('‚ùå ANDA TIDAK PUNYA HAK');
                return;
            }

            const idx = leaves.findIndex(l => l.id === id);
            if (idx !== -1) {
                leaves[idx].approved = false;
                leaves[idx].approvedBy = null;
                leaves[idx].approvedAt = null;
                saveData();
                showToast('‚ö†Ô∏è APPROVAL DIBATALKAN');
                updateAllUI();
            }
        }

        function deleteLeave(id) {
            if (confirm('APAKAH ANDA YAKIN INGIN MENGHAPUS?')) {
                leaves = leaves.filter(l => l.id !== id);
                saveData();
                showToast('üóëÔ∏è DATA DIHAPUS');
                updateAllUI();
            }
        }

        // ===== AUTHENTICATION =====
        function openApproverModal() {
            if (!currentUser) {
                document.getElementById('loginModal').classList.add('active');
            } else {
                document.getElementById('approverModal').classList.add('active');
                renderApprovers();
            }
        }

        function closeApproverModal() {
            document.getElementById('approverModal').classList.remove('active');
        }

        function closeLoginModal() {
            document.getElementById('loginModal').classList.remove('active');
            document.getElementById('loginEmail').value = '';
        }

        function login() {
            const email = document.getElementById('loginEmail').value.trim();
            if (!email) {
                showToast('‚ùå MASUKKAN EMAIL');
                return;
            }
            currentUser = email.toLowerCase();
            sessionStorage.setItem(USER_KEY, currentUser);
            closeLoginModal();
            updateAllUI();
            showToast(`‚úÖ LOGIN BERHASIL - ${email}`);
        }

        function logout() {
            currentUser = null;
            sessionStorage.removeItem(USER_KEY);
            updateAllUI();
            showToast('üö™ LOGOUT BERHASIL');
        }

        function toggleUserMenu() {
            // placeholder for menu toggle
        }

        // ===== APPROVER MANAGEMENT =====
        function renderApprovers() {
            const list = document.getElementById('approversList');
            if (approvers.length === 0) {
                list.innerHTML = '<p style="color: var(--text-secondary); font-size: 13px;">BELUM ADA APPROVER</p>';
                return;
            }

            list.innerHTML = approvers.map(email => `
                <div style="background: rgba(0, 255, 255, 0.15); padding: 8px 14px; border-radius: 2px; border: 1px solid var(--cyan); display: flex; align-items: center; gap: 8px; font-size: 13px;">
                    ${email}
                    <button type="button" onclick="removeApprover('${email}')" style="background: none; border: none; color: var(--danger); cursor: pointer; font-size: 16px;">√ó</button>
                </div>
            `).join('');
        }

        function addApprover() {
            const email = document.getElementById('newApproverEmail').value.trim();
            if (!email) {
                showToast('‚ùå MASUKKAN EMAIL');
                return;
            }
            if (approvers.length >= 2) {
                showToast('‚ùå MAKSIMAL 2 APPROVER');
                return;
            }
            if (approvers.includes(email.toLowerCase())) {
                showToast('‚ö†Ô∏è EMAIL SUDAH ADA');
                return;
            }
            approvers.push(email.toLowerCase());
            document.getElementById('newApproverEmail').value = '';
            renderApprovers();
        }

        function removeApprover(email) {
            approvers = approvers.filter(a => a !== email);
            renderApprovers();
        }

        function saveApprovers() {
            saveData();
            closeApproverModal();
            showToast('‚úÖ APPROVER TERSIMPAN');
        }

        // ===== UI RENDERING =====
        function updateAllUI() {
            updateUserDisplay();
            updateDivisionStatus();
            renderLeaves();
            updateCounts();
        }

        function updateUserDisplay() {
            const avatar = document.getElementById('userAvatar');
            const name = document.getElementById('userName');
            const logoutBtn = document.getElementById('logoutBtn');

            if (currentUser) {
                avatar.textContent = currentUser[0].toUpperCase();
                name.style.display = 'inline';
                name.textContent = currentUser.toUpperCase();
                logoutBtn.style.display = 'inline-block';
            } else {
                avatar.textContent = '-';
                name.style.display = 'none';
                logoutBtn.style.display = 'none';
            }
        }

        function updateDivisionStatus() {
            const month = document.getElementById('viewMonth').value;
            const html = Object.entries(DIVISION_LIMITS).map(([div, limit]) => {
                const count = leaves.filter(l => l.month === month && l.division === div && l.approved).length;
                const isFull = count >= limit;
                return `
                    <div class="division-card ${isFull ? 'full' : ''}">
                        <div class="division-name">${div}</div>
                        <div class="division-count">${count}/${limit}</div>
                        <div class="division-limit">${isFull ? 'üî¥ PENUH' : `${limit - count} TERSEDIA`}</div>
                    </div>
                `;
            }).join('');
            document.getElementById('divisionStatus').innerHTML = html;
        }

        function renderLeaves() {
            const pending = leaves.filter(l => !l.approved).sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));
            const approved = leaves.filter(l => l.approved).sort((a, b) => new Date(b.approvedAt || b.createdAt) - new Date(a.approvedAt || a.createdAt));

            document.getElementById('pendingList').innerHTML = renderLeaveList(pending);
            document.getElementById('approvedList').innerHTML = renderLeaveList(approved);
            document.getElementById('allList').innerHTML = renderLeaveList(leaves.sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt)));
        }

            if (list.length === 0) {
                return '<div class="empty-state"><div class="empty-state-text">TIDAK ADA DATA</div></div>';
            }

            return list.map(leave => {
                const isApprover = currentUser && approvers.includes(currentUser.toLowerCase());

                const endDate = new Date(leave.endDate).toLocaleDateString('id-ID', { day: 'numeric', month: 'long', year: 'numeric' });
                const days = Math.ceil((new Date(leave.endDate) - new Date(leave.startDate)) / (1000 * 60 * 60 * 24)) + 1;

                return `

                            </div>

                            </span>
                        </div>

                        <div style="padding: 16px; background: rgba(0, 0, 0, 0.3); border: 1px solid var(--magenta); border-radius: 2px; margin-bottom: 16px;">
                            <div style="margin-bottom: 12px; padding-bottom: 12px; border-bottom: 1px solid var(--magenta);">

                                <div style="font-size: 14px; color: var(--text-primary); font-weight: 600;">${leave.perihal}</div>
                            </div>

                            <div style="display: grid; grid-template-columns: 1fr 1fr; gap: 12px; margin-bottom: 12px; padding-bottom: 12px; border-bottom: 1px solid var(--magenta);">

                                    <div>

                                        <div style="font-size: 13px; color: var(--text-primary);">${leave.situs}</div>
                                    </div>
                                ` : `<div></div>`}
                                ${leave.noPaspor ? `
                                    <div>
                                        <div style="font-size: 11px; color: var(--cyan); font-weight: 700; letter-spacing: 0.1em; text-transform: uppercase; text-shadow: 0 0 5px var(--cyan); margin-bottom: 4px;">NO PASPOR</div>
                                        <div style="font-size: 13px; color: var(--text-primary);">${leave.noPaspor}</div>

        function updateCounts() {
            const pending = leaves.filter(l => !l.approved).length;
            const approved = leaves.filter(l => l.approved).length;
            document.getElementById('pendingCount').textContent = pending;
            document.getElementById('approvedCount').textContent = approved;
            document.getElementById('allCount').textContent = leaves.length;
        }

        function switchTab(tab) {
            document.querySelectorAll('.tab-trigger').forEach(t => t.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));

            event.target.classList.add('active');
            document.getElementById(tab + '-tab').classList.add('active');
        }

        // ===== NOTIFICATIONS =====
        function showToast(message) {
            const toast = document.createElement('div');
            toast.className = 'toast';
            toast.textContent = message;
            document.body.appendChild(toast);
            setTimeout(() => {
                toast.style.animation = 'toastSlideIn 0.3s ease reverse';
                setTimeout(() => toast.remove(), 300);
            }, 2500);
        }
